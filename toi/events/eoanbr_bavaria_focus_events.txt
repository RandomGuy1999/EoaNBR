				add_country_leader_role = {
					promote_leader = yes

						desc = "POLITICS_GENERIC_EOANB_DESC"

			}

			set_country_flag = bavaria_maximilian_ii_is_dead

	option = {
		ai_chance = { factor = 100 }
		name = bavariafocus.54.a
