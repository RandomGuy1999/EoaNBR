country_event = { #TCS Fails Revolt
	id = trucial.16
	title = trucial.16.t
	desc = trucial.16.d
		
	fire_only_once = yes
    is_triggered_only = yes
	
	option = { #Annex Trucial
	    name = trucial.16.a
		# log = "option:	    name = trucial.16.a"
		ai_chance = { factor = 50 }
		ENG = {
			annex_country = {
				target = TCS
				transfer_troops = no
			}
		}
		add_political_power = -200
	}

	option = { #Compromise
	    name = trucial.16.b
		# log = "option:	    name = trucial.16.b"
		ai_chance = { factor = 50 }
		ENG = {
			set_autonomy = {
				target = TCS
				autonomy_state = autonomy_british_self_gov
			}
		}
		add_political_power = -50
		TCS = {
			set_cosmetic_tag = TCS_UK_D
			add_ideas = AFR_british_colony
		}
	}
}
