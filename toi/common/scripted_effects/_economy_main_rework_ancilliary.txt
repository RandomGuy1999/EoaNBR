economy_count_iron_mines_effect = {
	set_temp_variable = { x_temp = 0 }
	every_owned_state = {
		limit = {
			# has_variable = a
			has_resources_amount = {
				resource = iron
				amount > 0
			}
		}
		add_to_temp_variable = { x_temp = resource@iron }
		log = "[THIS.getName] has [?THIS.resource@iron]"
	}
	
	log = "[?x_temp] total iron mines"
}
