afr_call_state_levy_units = {
	if = {
		limit = {
			OWNER = {
				OR = {
					has_idea = afr_large_kingdom
					has_idea = afr_reforming_kingdom
					has_idea = Oceania_eastern_oriented_regional_power
					has_idea = Oceania_western_oriented_regional_power
				}
			}
		}
		if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 6 compare = greater_than_or_equals } }
				THIS = {
					food > 3
				}
			}
			set_temp_variable = { Temp_State_farm_level = 4 }
			add_to_temp_variable = { State_levy_army_num = 8000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 6 compare = greater_than_or_equals } }
				THIS = {
					food > 2
				}
			}
			set_temp_variable = { Temp_State_farm_level = 3 }
			add_to_temp_variable = { State_levy_army_num = 6000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 6 compare = greater_than_or_equals } }
				THIS = {
					food > 1
				}
			}
			set_temp_variable = { Temp_State_farm_level = 2 }
			add_to_temp_variable = { State_levy_army_num = 4000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 6 compare = greater_than_or_equals } }
				THIS = {
					food > 0
				}
			}
			set_temp_variable = { Temp_State_farm_level = 1 }
			add_to_temp_variable = { State_levy_army_num = 2000 }
		}
		if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 9 compare = greater_than_or_equals } }
				THIS = { state_population > 1000000 }
			}
			add_to_temp_variable = { State_levy_army_num = 12000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 6 compare = greater_than_or_equals } }
				THIS = { state_population > 1000000 }
			}
			add_to_temp_variable = { State_levy_army_num = 11000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 9 compare = greater_than_or_equals } }
				THIS = { state_population < 1000000 }
				THIS = { state_population > 500000 }
			}
			add_to_temp_variable = { State_levy_army_num = 10000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 6 compare = greater_than_or_equals } }
				THIS = { state_population < 1000000 }
				THIS = { state_population > 500000 }
			}
			add_to_temp_variable = { State_levy_army_num = 9000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 9 compare = greater_than_or_equals } }
				THIS = { state_population < 500000 }
				THIS = { state_population > 250000 }
			}
			add_to_temp_variable = { State_levy_army_num = 8000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 6 compare = greater_than_or_equals } }
				THIS = { state_population < 500000 }
				THIS = { state_population > 250000 }
			}
			add_to_temp_variable = { State_levy_army_num = 7000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 9 compare = greater_than_or_equals } }
				THIS = { state_population > 250000 }
				THIS = { state_population < 100000 }
			}
			add_to_temp_variable = { State_levy_army_num = 6000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 6 compare = greater_than_or_equals } }
				THIS = { state_population > 250000 }
				THIS = { state_population < 100000 }
			}
			add_to_temp_variable = { State_levy_army_num = 5000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 9 compare = greater_than_or_equals } }
				THIS = { state_population > 100000 }
				THIS = { state_population < 50000 }
			}
			add_to_temp_variable = { State_levy_army_num = 4000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 6 compare = greater_than_or_equals } }
				THIS = { state_population > 100000 }
				THIS = { state_population < 50000 }
			}
			add_to_temp_variable = { State_levy_army_num = 3000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 9 compare = greater_than_or_equals } }
				THIS = { state_population < 50000 }
			}
			add_to_temp_variable = { State_levy_army_num = 2000 }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 6 compare = greater_than_or_equals } }
				THIS = { state_population < 50000 }
			}
			add_to_temp_variable = { State_levy_army_num = 1000 }
		}
		if = {
			limit = {
				check_variable = {
					var = State_levy_army_num
					value = 12000
					compare = greater_than
				}
			}
			set_temp_variable = { State_levy_army_num = 12000 }
		}
		subtract_from_temp_variable = { State_levy_army_num = State_levy_army_used }
		if = {
			limit = {
				check_variable = {
					var = State_levy_army_num
					value = 3000
					compare = greater_than_or_equals
				}
			}
			set_temp_variable = {temp_state_levy_army = State_levy_army_num}
			divide_temp_variable = {temp_state_levy_army = 3000}
			round_variable = temp_state_levy_army
			add_to_temp_array = {
				array = num_of_state_levy_divisions_to_raise
				value = 1
				index = 1
			}
			resize_temp_array = {num_of_state_levy_divisions_to_raise = temp_state_levy_army}
			if = {
				limit = { OWNER = { NOT = { has_template = "Levies" } } }
				OWNER = {
					division_template = {
						name = "Levies"
						priority = 1
						division_names_group = AFR_STATE_LEVY_01
						regiments = {
							feudal_infantry = { x = 0 y = 0 }
							feudal_infantry = { x = 0 y = 1 }
							feudal_infantry = { x = 0 y = 2 }
						}
						is_locked = yes
					}
				}
			}
			if = {
				limit = { OWNER = { NOT = { has_template = "Levy Cavalry" } } }
				OWNER = {
					division_template = {
						name = "Levy Cavalry"
						priority = 1
						division_names_group = AFR_STATE_LEVY_01
						regiments = {
							state_levy_cavalry = { x = 0 y = 0 }
							state_levy_cavalry = { x = 0 y = 1 }
							state_levy_cavalry = { x = 0 y = 2 }
						}
						is_locked = yes
					}
				}
			}
			for_each_loop = {
				array = num_of_state_levy_divisions_to_raise
				random_list = {
					25 = {
						create_unit = {
							division = "name = \"Levies\" division_template = \"Levies\" start_experience_factor = 0.60 start_equipment_factor = 1"
							owner = PREV
						}
					}
					75 = {
						create_unit = {
							division = "name = \"Levy Cavalry\" division_template = \"Levy Cavalry\" start_experience_factor = 0.60 start_equipment_factor = 1"
							owner = PREV
						}
						modifier = {
							factor = 0
							PREV = { NOT = { has_country_flag = afr_should_call_state_levy_mixed_flag } }
						}
					}
				}
				subtract_from_temp_variable = { State_levy_army_num = 3000 }
				clamp_variable = {
					var = State_levy_army_num
					min = 0
					max = 12000
				}
				add_to_variable = { State_levy_army_used = 3000 }
				clamp_variable = {
					var = State_levy_army_used
					min = 0
					max = 12000
				}
			}
		}
	}
}

afr_empire_mechanic_weekly_pulse_check = {
	every_state = {
		limit = { OWNER = { has_variable = has_afr_empire_mechanic_system } }
		###################
		##  Tax Effects  ##
		###################
		if = {
			limit = { OWNER = { has_idea = very_high_taxes } }
			afr_empire_mechanic_tax_prosperity_check_very_high = yes
		}
		else_if = {
			limit = { OWNER = { has_idea = high_taxes } }
			afr_empire_mechanic_tax_prosperity_check_high = yes
		}
		else_if = {
			limit = { OWNER = { has_idea = normal_taxes } }
			afr_empire_mechanic_tax_prosperity_check_normal = yes
		}
		else_if = {
			limit = { OWNER = { has_idea = low_taxes } }
			afr_empire_mechanic_tax_prosperity_check_low = yes
		}
		else_if = {
			limit = { OWNER = { has_idea = very_low_taxes } }
			afr_empire_mechanic_tax_prosperity_check_very_low = yes
		}
		#############
		####Angry####
		#############
		if = {
			limit = {
				check_variable = { var = afr_empire_state_public_opinion_var value = 0 compare = greater_than_or_equals }
				check_variable = { var = afr_empire_state_public_opinion_var value = 20 compare = less_than_or_equals }
			}
			afr_empire_state_public_opinion_var_weekly_calculation = yes
			afr_empire_state_public_opinion_var_increase_by_3 = yes
		}
		##################
		####Discontent####
		##################
		else_if = {
			limit = {
				check_variable = { var = afr_empire_state_public_opinion_var value = 20 compare = greater_than }
				check_variable = { var = afr_empire_state_public_opinion_var value = 30 compare = less_than_or_equals }
			}
			afr_empire_state_public_opinion_var_weekly_calculation = yes
			afr_empire_state_public_opinion_var_increase_by_2 = yes
		}
		###############
		####Content####
		###############
		else_if = {
			limit = {
				check_variable = { var = afr_empire_state_public_opinion_var value = 30 compare = greater_than }
				check_variable = { var = afr_empire_state_public_opinion_var value = 50 compare = less_than }
			}
			afr_empire_state_public_opinion_var_weekly_calculation = yes
			afr_empire_state_public_opinion_var_increase_by_1 = yes
		}
		else_if = {
			limit = {
				check_variable = { var = afr_empire_state_public_opinion_var value = 50 compare = equals }
			}
			afr_empire_state_public_opinion_var_weekly_calculation = yes
		}
		else_if = {
			limit = {
				check_variable = { var = afr_empire_state_public_opinion_var value = 50 compare = greater_than }
				check_variable = { var = afr_empire_state_public_opinion_var value = 70 compare = less_than }
				}
			afr_empire_state_public_opinion_var_weekly_calculation = yes
			afr_empire_state_public_opinion_var_decrease_by_1 = yes
		}
		#############
		####Happy####
		#############
		else_if = {
			limit = {
				check_variable = { var = afr_empire_state_public_opinion_var value = 70 compare = greater_than_or_equals }
				check_variable = { var = afr_empire_state_public_opinion_var value = 80 compare = less_than }
				}
			afr_empire_state_public_opinion_var_weekly_calculation = yes
			afr_empire_state_public_opinion_var_decrease_by_2 = yes
		}
		##############
		####Elated####
		##############
		else_if = {
			limit = {
				check_variable = { var = afr_empire_state_public_opinion_var value = 80 compare = greater_than_or_equals }
				check_variable = { var = afr_empire_state_public_opinion_var value = 100 compare = less_than_or_equals }
			}
			afr_empire_state_public_opinion_var_weekly_calculation = yes
			afr_empire_state_public_opinion_var_decrease_by_3 = yes
		}
	}
}
afr_empire_mechanic_monthly_pulse_check = {
	every_state = {
		limit = { OWNER = { has_variable = has_afr_empire_mechanic_system } }
		################################
		#  Public Opinion Pulse Check  #
		################################
		if = {
			limit = {
				check_variable = { var = afr_empire_state_public_opinion_var value = 0 compare = greater_than_or_equals }
				check_variable = { var = afr_empire_state_public_opinion_var value = 15 compare = less_than_or_equals }
			}
			afr_empire_state_loyalty_var_decrease_by_1 = yes
		}
		else_if = {
			limit = {
				check_variable = { var = afr_empire_state_public_opinion_var value = 15 compare = greater_than }
				check_variable = { var = afr_empire_state_public_opinion_var value = 30 compare = less_than_or_equals }
			}
			afr_empire_state_loyalty_var_decrease_by_half = yes
		}
		else_if = {
			limit = {
				check_variable = { var = afr_empire_state_public_opinion_var value = 30 compare = greater_than }
				check_variable = { var = afr_empire_state_public_opinion_var value = 70 compare = less_than }
			}
		}
		else_if = {
			limit = {
				check_variable = { var = afr_empire_state_public_opinion_var value = 70 compare = greater_than_or_equals }
				check_variable = { var = afr_empire_state_public_opinion_var value = 85 compare = less_than }
			}
			afr_empire_state_loyalty_var_increase_by_half = yes
		}
		else_if = {
			limit = {
				check_variable = { var = afr_empire_state_public_opinion_var value = 85 compare = greater_than_or_equals }
				check_variable = { var = afr_empire_state_public_opinion_var value = 100 compare = less_than_or_equals }
			}
			afr_empire_state_loyalty_var_increase_by_1 = yes
		}
		###############################
		#  State Opinion Pulse Check  #
		###############################
		if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 0 compare = less_than } }
			}
			OWNER = { afr_empire_state_loyalty_full_core_event_failed_effect = yes }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 0 compare = greater_than_or_equals } }
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 3 compare = less_than } }
			}
			OWNER = { afr_empire_state_loyalty_full_core_event_failed_effect = yes }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 3 compare = greater_than_or_equals } }
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 6 compare = less_than } }
			}
			OWNER = { afr_empire_state_loyalty_full_core_event_failed_effect = yes }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 6 compare = greater_than_or_equals } }
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 9 compare = less_than } }
			}
			OWNER = { afr_empire_state_loyalty_full_core_event_failed_effect = yes }
		}
		else_if = {
			limit = {
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 9 compare = greater_than_or_equals } }
				OWNER = { check_variable = { var = PREV.afr_empire_state_loyalty_var@THIS value = 10 compare = less_than_or_equals } }
			}

		}
	}

afr_empire_politics_state_manager_martial_law_idea_effect = {
	hidden_effect = {
		if = {
			limit = { has_variable = PREV.afr_empire_politics_state_manager_martial_law_enabled@THIS }
			add_to_variable = { var = afr_empire_politics_state_manager_martial_law_var value = -0.33 }
		}
		else = {
			add_to_variable = { var = afr_empire_politics_state_manager_martial_law_var value = 0.33 }
		}
		if = {
			limit = { check_variable = { var = afr_empire_politics_state_manager_martial_law_var value = 0 compare = less_than} }
			set_variable = {
				var = afr_empire_politics_state_manager_martial_law_idea_var
				value = afr_empire_politics_state_manager_martial_law_var
			}
		}
		if = {
			limit = {
				check_variable = { var = afr_empire_politics_state_manager_martial_law_var value = 0 compare = less_than}
				NOT = { has_dynamic_modifier = { modifier = afr_empire_politics_state_manager_martial_law_idea } }
			}
			add_dynamic_modifier = { modifier = afr_empire_politics_state_manager_martial_law_idea }
		}
		else_if = {
			limit = {
				check_variable = { var = afr_empire_politics_state_manager_martial_law_var value = 0 compare = equals }
				has_dynamic_modifier = { modifier = afr_empire_politics_state_manager_martial_law_idea }
			}
			remove_dynamic_modifier = { modifier = afr_empire_politics_state_manager_martial_law_idea }
		}
	}
}
