PIE_swfi_escalate = {
	if = {
		limit = { has_global_flag = second_italo_austrian_war_flag }
		if = {
			limit = {
				check_variable = { PIE.ita_swfi_war_escalation_level_var = 0 }
				controls_state = 159
			}
			set_variable = {
				var = PIE.ita_swfi_war_escalation_level_var
				value = 1
			}
		}
		else_if = {
			limit = {
				check_variable = { PIE.ita_swfi_war_escalation_level_var = 1 }
				controls_state = 160
			}
			set_variable = {
				var = PIE.ita_swfi_war_escalation_level_var
				value = 2
			}
		}
		else_if = {
			limit = {
				check_variable = { PIE.ita_swfi_war_escalation_level_var = 2 }
				count_triggers = {
					amount = 2
					controls_state = 39
					controls_state = 736
					controls_state = 153
					controls_state = 1235
					controls_state = 152
					controls_state = 969
				}
			}
			set_variable = {
				var = PIE.ita_swfi_war_escalation_level_var
				value = 3
			}
		}
		else_if = {
			limit = {
				check_variable = { PIE.ita_swfi_war_escalation_level_var = 3 }
				has_global_flag = {
					flag = PIE_swfi_france_rejects_peace_flag
					value = 1
				}
			}
			set_variable = {
				var = PIE.ita_swfi_war_escalation_level_var
				value = 4
			}
		}
		else_if = {
			limit = {
				check_variable = { PIE.ita_swfi_war_escalation_level_var = 4 }
				has_global_flag = {
					flag = PIE_swfi_france_rejects_peace_flag
					value = 3
				}
			}
			set_variable = {
				var = PIE.ita_swfi_war_escalation_level_var
				value = 5
			}
		}
		PIE_swfi_map_update = yes
	}
}

PIE_swfi_map_update = {
	for_each_scope_loop = {
		array = PIE.swfi_state_map_array
		if = {
			limit = { is_controlled_by = FRA }
			set_variable = { swfi_state_map_frame = 4 }
			log = "[THIS.GetName] is controlled by France"
		}
		else_if = {
			limit = {
				controller = {
					OR = {
						tag = AUS
						is_ally_with = AUS
					}
				}
			}
			set_variable = { swfi_state_map_frame = 3 }
			log = "[THIS.GetName] is controlled by Austria"
		}
		else_if = {
			limit = {
				controller = {
					OR = {
						tag = PIE
						is_ally_with = PIE
					}
				}
			}
			set_variable = { swfi_state_map_frame = 2 }
			log = "[THIS.GetName] is controlled by SP"
		}
		else = {
			set_variable = { swfi_state_map_frame = 1 }
			log = "[THIS.GetName] is controlled by Italian Minor"
		}
	}
}
