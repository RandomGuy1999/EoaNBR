	allowed = {
		OR={
			original_tag = FRA
			original_tag = ENG
			original_tag = SPR
		}
	}

	enable = {
		# tag = QNG
		has_war_with = MEX

		}
	}

	ai_strategy = {
		type = invade
		id = "MEX"
		value = 200
	}

	ai_strategy = {
		type = antagonize
		id = "MEX"
		value = 200
	}
}
